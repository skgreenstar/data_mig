-- 1. 가격 테이블 생성
CREATE TABLE IF NOT EXISTS main.strategic_syk.prices (
    date        DATE,
    grade       STRING,
    price_usd_bbl DOUBLE,
    source      STRING,
    collected_at TIMESTAMP
) USING DELTA;

-- 2. 뉴스 테이블 생성
CREATE TABLE IF NOT EXISTS main.strategic_syk.news (
    id           STRING,
    published_at TIMESTAMP,
    title        STRING,
    source       STRING,
    category     STRING,
    sentiment    DOUBLE,
    impact_score DOUBLE,
    url          STRING,
    summary      STRING,
    collected_at TIMESTAMP,
    title_ko     STRING
) USING DELTA;

-- 3. 거시 지표 테이블 생성
CREATE TABLE IF NOT EXISTS main.strategic_syk.macro (
    date        DATE,
    metric      STRING,
    value       DOUBLE,
    source      STRING,
    collected_at TIMESTAMP
) USING DELTA;
