-- 1. 가격 테이블 생성
CREATE TABLE IF NOT EXISTS main.strategic_syk.prices (
    date        DATE,
    grade       STRING,
    price_usd_bbl DOUBLE,
    source      STRING,
    collected_at TIMESTAMP
) USING DELTA;

-- 2. 뉴스 테이블 생성
CREATE TABLE IF NOT EXISTS main.strategic_syk.news (
    id           STRING,
    published_at TIMESTAMP,
    title        STRING,
    source       STRING,
    category     STRING,
    sentiment    DOUBLE,
    impact_score DOUBLE,
    url          STRING,
    summary      STRING,
    collected_at TIMESTAMP,
    title_ko     STRING
) USING DELTA;

-- 3. 거시 지표 테이블 생성
CREATE TABLE IF NOT EXISTS main.strategic_syk.macro (
    date        DATE,
    metric      STRING,
    value       DOUBLE,
    source      STRING,
    collected_at TIMESTAMP
) USING DELTA;


-- 1. 가격 데이터 (price_usd_bbl 매핑 및 현재시각 추가)
INSERT INTO main.strategic_syk.prices (date, grade, price_usd_bbl, source, collected_at)
SELECT date, grade, price_usd_bbl, source, current_timestamp() 
FROM json.`/Volumes/main/strategic_syk/migration_files/prices.jsonl`;

-- 2. 뉴스 데이터
INSERT INTO main.strategic_syk.news (id, published_at, title, source, category, sentiment, impact_score, url, summary, title_ko, collected_at)
SELECT id, published_at, title, source, category, sentiment, impact_score, url, summary, title_ko, current_timestamp()
FROM json.`/Volumes/main/strategic_syk/migration_files/news.jsonl`;

-- 3. 거시 지표 데이터
INSERT INTO main.strategic_syk.macro (date, metric, value, source, collected_at)
SELECT date, metric, value, source, current_timestamp()
FROM json.`/Volumes/main/strategic_syk/migration_files/macro.jsonl`;
